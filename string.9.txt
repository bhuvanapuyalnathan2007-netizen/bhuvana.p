#include<stdio.h>
#include<string.h>
int main() {
    char a[20];
    int i,len,flag=1;
    scanf("%s",a);
    len=strlen(a);
    for(i=0;i<len/2;i++) {
        if(a[i]!=a[len-1-i]){
            flag=0;
            break;
        }
    }
    if(flag) {
        printf("polindrome");
    }else {
        printf("not polindrome");
        return 0;
    }
}
