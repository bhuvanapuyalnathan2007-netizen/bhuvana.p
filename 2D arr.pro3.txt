#include<stdio.h>
int main() {
    int r,c;
    printf("Enter the value of r and c\n");
    scanf("%d %d",&r,&c);
    printf("Enter the value for the array element:\n");
    int arr[r][c];
    for (int i = 0; i < r; i++) {
        for (int j = 0; j < c; j++) {
            scanf("%d",&arr[i][j]);
        }
        int count;

        for (int i = 0; i < c; i++) {
            for (int j = 0; j < r; j++) {
                if (arr[i][j] == 0 ){
                count++;
                }
            }
        }
        printf("%d number of seats we had empty",count);}
        return 0;
}
