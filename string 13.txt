#include<stdio.h>
#include<string.h>
int main() {
    char address[100];
    int count = 0;
    fgets(address,sizeof(address),stdin);
    for (int i = 0; address[i] != '\0'; i++) {
        if (address[i] == ' ') {
            count++;
        }
    }
    printf("%d\n",count);
    return 0;
}
