#include<stdio.h>
int main() {
    int r,c;
    printf("enter the value of r and c\n");
    scanf("%d %d",&r,&c);
    int arr[r][c];
    printf("enter the array elements\n");
    for(int i=0;i<r;i++) {
        for(int j=0;j<c;j++) {
            scanf("%d",&arr[i][j]);
        }
    }
    for(int i=0;i<r;i++) {
        int max=0;
        for(int j=0;j<c;j++) {
            if(arr[i][j]>max) {
                max=arr[i][j];
            }

        }
        printf("%d is the highest temp row\n",i+1 );
    }
    return 0;
}
