#include<stdio.h>
int main(){
    int size,count=0;
    scanf("%d",&size);
    int arr[size];
    for(int i=0;i<size;i++)
    {
        scanf("%d",&arr[i]);
        for(int j=0;j<size;j++)
        {
            if (arr[j]>=100){
                count++;
            }
        }

            return 0;
        }
    }
