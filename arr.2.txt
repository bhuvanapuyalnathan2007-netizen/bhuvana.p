#include<stdio.h>
int main()
{
    int size;
    scanf("%d",&size);
    int arr[size];
    for(int i=0;i<size;i++)
    {
        scanf("%d",&arr[i]);
    }
    int count=0;
    for(int j=0;j<size;j++)
    {
        if(arr[j]==1)
        {
            count++;
        }
        printf("%d number of presengt student",count);
        return 0;
    }

}
