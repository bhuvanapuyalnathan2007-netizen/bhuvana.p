#include<stdio.h>
int main() {
    int r,c;
    printf("Enter the value of r and c\n");
    scanf("%d%d",&r,&c);
    int arr[r][c];
    for(int i=0;i<r;i++) {
        for(int j=0;j<c;j++) {
            scanf("%d",&arr[i][j]);
        }
    }
    int count=0;
    for(int i=0;i<r;i++) {
        for(int j=0;j<c;j++) {
            if (arr[i][j]==200) {
                count++;
            }
        }
    }
    printf("%d number of seats we had empty",count);
    return 0;

}
