#include<stdio.h>
int reversenumber(int n)
{
   int reverse=0;
   int rem;
   while(n!=0){
      rem=n%10;
      n=n/10;
      reverse=reverse*10+rem;
   }
   return reverse;
}
int main() {
   int n,result;
   scanf("%d",&n);
   result=reversenumber(n);
   printf("%d",result);
   return 0;
}
