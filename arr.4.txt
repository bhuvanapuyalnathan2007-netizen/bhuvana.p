#include<stdio.h>
int main()
{
    int size,count=0;
    scanf("%d",&size);
    int arr[size];
    for(int i=0;i<size;i++)
    {
        scanf("%d",&arr[i]);
    }
    int sum=0;
    for(int j=0;j<size;j++)
    {
        sum=sum+arr[j];
        int avg=sum/size;
        printf("%d",avg);
        for(int k=0;k<size;k++)
        {
            if (arr[k]<avg){
                count++;
            }
            printf("%d that is greater than avg",count);
            return 0;
        }
    }
}
