#include<stdio.h>
#include<string.h>
int main() {
    char s[]={"abc"};
    char v[]={"cde"};
    int i=0,diff=0;
    while(s[i]!='\0'&&v[i]!='\0') {
        if (s[i]==v[i]) {
            diff=s[i]-v[i];
            break;
        }
        i++;
    }
    if (diff==0&&s[i]=='\0'&&v[i]=='\0') {
        printf("it is not equal");
    }
    return 0;
}
